def Upwind(a, u0, D, N):
    '''
    Solves the one-dimensional, linear, hyperbolic PDE u_t + a*u_x = 0
    with initial value function u0(x) and periodic boundary values
    using the upwind scheme.
    
    returns 2D-array of function values
    
    a .... constant
    u0 ... intial value function in the x variable
    D .... [[a, b], [c, d]] array of corners of the square
    N .... (N_x, N_t) is a tuple of discretization steps for x and t respectively
    '''
    
    D = np.array(D)
    hx = (D[0, 1] - D[0, 0]) / N[0]
    ht = (D[1, 1] - D[1, 0]) / N[1]
    u = np.zeros((N[0]+1, N[1]+1))
    
    # initial values
    u[:, 0] = u0(D[0, 0] + hx * np.arange(N[0] + 1))
    
    if a <= 0:
        for j in range(N[1]):
            for i in range(N[0]):
                u[i, j+1] = u[i, j] - a * ht / hx * (u[i+1, j] - u[i, j])
            u[N[0], j+1] = u[0, j+1]
    else:
        for j in range(N[1]):
            for i in range(1, N[0]+1):
                u[i, j+1] = u[i, j] - a * ht / hx * (u[i, j] - u[i-1, j])
            u[0, j+1] = u[N[0], j+1]
    
    return u

def LaxFriedrich(a, u0, D, N):
    '''
    Solves the one-dimensional, linear, hyperbolic PDE u_t + a*u_x = 0
    with initial value function u0(x) and periodic boundary values
    using the Lax-Friedrichs method.
    
    returns 2D-array of function values
    
    a .... constant
    u0 ... intial value function in the x variable
    D .... [[a, b], [c, d]] array of corners of the square
    N .... (N_x, N_t) is a tuple of discretization steps for x and t respectively
    '''
    
    D = np.array(D)
    hx = (D[0, 1] - D[0, 0]) / N[0]
    ht = (D[1, 1] - D[1, 0]) / N[1]
    u = np.zeros((N[0]+1, N[1]+1))
    
    # initial values
    u[:, 0] = u0(D[0, 0] + hx * np.arange(N[0] + 1))
    
    for j in range(N[1]):
        for i in range(1, N[0]):
            u[i, j+1] = (u[i+1, j] + u[i-1, j]) / 2 - a * ht * (u[i+1, j] - u[i-1, j]) / (2 * hx)
        u[0, j+1] = u[N[0], j+1] = (u[1, j] + u[N[0]-1, j]) / 2 - a * ht * (u[1, j] - u[N[0]-1, j]) / (2 * hx)
    
    return u

a = -1
D = np.array([[0, 1], [0, 1]])

def u0(x):
    return np.sin(2 * np.pi * x)

def exact(x, t):
    return u0(x - a * t)

def error(exact_vals, approx_vals):
    '''Maxmimum norm'''
    return max(np.abs(exact_vals.flatten() - approx_vals.flatten()))

def order(err_old, err_new, step_old, step_new):
    '''Approximation of order of convergence'''
    return np.log(err_new / err_old) / np.log(step_new / step_old)

def OUTPUT(Ns, T, c=2):
    # c > 1 to satisfy Courant-Friedrichs-Lewy condition
    
    print("\t\t Upwind | Lax-Friedrich")
    
    U_errors = ()
    L_errors = ()
    hs = ()
    
    for N in Ns:
        M = (N, c*N)
        U = Upwind(a, u0, D, M)
        L = LaxFriedrich(a, u0, D, M)
        h = (D[0, 1] - D[0, 0]) / N
        x = D[0, 0] + h * np.arange(N+1)
        i = int(M[1] * (T - D[1, 0]) / (D[1, 1] - D[1, 0]))
        exact_vals = exact(x, T)
        U_err = error(exact_vals, U[:, i])
        L_err = error(exact_vals, L[:, i])
        U_errors += (U_err,)
        L_errors += (L_err,)
        hs += (h,)
        
        print(f"error, N = {N:3.0f}:\t {U_err:6.4f} | {L_err:13.4f}")
        
        plt.plot(x, U[:, i], label=f"N={N}, Upwind")
        plt.plot(x, L[:, i], label=f"N={N}, Lax-Friedrich")
        
    U_ord = order(U_errors[0], U_errors[-1], hs[0], hs[-1])
    L_ord = order(L_errors[0], L_errors[-1], hs[0], hs[-1])
    
    print(f"order of conv.:\t {U_ord:6.0f} | {L_ord:13.0f}")
    
    x = np.linspace(D[0, 0], D[0, 1], 101)
    plt.plot(x, exact(x, T), "dimgray", label="exact")
    plt.legend()
    plt.title(f"T = {T}, c = {c}")
    plt.show()
