def Upwind(a, u0, D, N):
    '''
    Solves the one-dimensional, linear, hyperbolic PDE u_t + a*u_x = 0
    with initial value function u0(x) and periodic boundary values
    using the upwind scheme.
    
    returns 2D-array of function values
    
    a .... constant
    u0 ... intial value function in the x variable
    D .... [[a, b], [c, d]] array of corners of the square
    N .... (N_x, N_t) is a tuple of discretization steps for x and t respectively
    '''
    
    D = np.array(D)
    hx = (D[0, 1] - D[0, 0]) / N[0]
    ht = (D[1, 1] - D[1, 0]) / N[1]
    u = np.zeros((N[0]+1, N[1]+1))
    
    # initial values
    u[:, 0] = u0(D[0, 0] + hx * np.arange(N[0] + 1))
    
    if a <= 0:
        for j in range(N[1]):
            for i in range(N[0]):
                u[i, j+1] = u[i, j] - a * ht / hx * (u[i+1, j] - u[i, j])
            u[N[0], j+1] = u[0, j+1]
    else:
        for j in range(N[1]):
            for i in range(1, N[0]+1):
                u[i, j+1] = u[i, j] - a * ht / hx * (u[i, j] - u[i-1, j])
            u[0, j+1] = u[N[0], j+1]
    
    return u

def LaxFriedrich(a, u0, D, N):
    '''
    Solves the one-dimensional, linear, hyperbolic PDE u_t + a*u_x = 0
    with initial value function u0(x) and periodic boundary values
    using the Lax-Friedrichs method.
    
    returns 2D-array of function values
    
    a .... constant
    u0 ... intial value function in the x variable
    D .... [[a, b], [c, d]] array of corners of the square
    N .... (N_x, N_t) is a tuple of discretization steps for x and t respectively
    '''
    
    D = np.array(D)
    hx = (D[0, 1] - D[0, 0]) / N[0]
    ht = (D[1, 1] - D[1, 0]) / N[1]
    u = np.zeros((N[0]+1, N[1]+1))
    
    # initial values
    u[:, 0] = u0(D[0, 0] + hx * np.arange(N[0] + 1))
    
    for j in range(N[1]):
        for i in range(1, N[0]):
            u[i, j+1] = (u[i+1, j] + u[i-1, j]) / 2 - a * ht * (u[i+1, j] - u[i-1, j]) / (2 * hx)
        u[0, j+1] = u[N[0], j+1] = (u[1, j] + u[N[0]-1, j]) / 2 - a * ht * (u[1, j] - u[N[0]-1, j]) / (2 * hx)
    
    return u

a = -1
D = np.array([[0, 1], [0, 1]])
N1 = 50
N2 = 400
c = 2  # c > 1 to satisfy Courant-Friedrichs-Lewy condition
M1 = (N1, c*N1)
M2 = (N2, c*N2)

def u0(x):
    return np.sin(2 * np.pi * x)

def exact(x, t):
    return u0(x - a * t)

def error(exact_vals, approx_vals):
    '''Maxmimum norm'''
    return max(np.abs(exact_vals.flatten() - approx_vals.flatten()))

def order(err_old, err_new, step_old, step_new):
    '''Approximation of order of convergence'''
    return np.log(err_new / err_old) / np.log(step_new / step_old)

def OUTPUT(T):
    U1 = Upwind(a, u0, D, M1)
    U2 = Upwind(a, u0, D, M2)
    L1 = LaxFriedrich(a, u0, D, M1)
    L2 = LaxFriedrich(a, u0, D, M2)
    h1 = (D[0, 1] - D[0, 0]) / N1
    h2 = (D[0, 1] - D[0, 0]) / N2
    x1 = D[0, 0] + h1 * np.arange(N1+1)
    x2 = D[0, 0] + h2 * np.arange(N2+1)
    i1 = int(M1[1] * (T - D[1, 0]) / (D[1, 1] - D[1, 0]))
    i2 = int(M2[1] * (T - D[1, 0]) / (D[1, 1] - D[1, 0]))
    exact_vals1 = exact(x1, T)
    exact_vals2 = exact(x2, T)
    
    eu1 = error(exact_vals1, U1[:, i1])
    eu2 = error(exact_vals2, U2[:, i2])
    el1 = error(exact_vals1, L1[:, i1])
    el2 = error(exact_vals2, L2[:, i2])
    ou = order(eu1, eu2, h1, h2)
    ol = order(el1, el2, h1, h2)
    print("\t\t Upwind | Lax-Friedrich")
    print(f"error, N = {N1:3.0f}:\t {eu1:6.4f} | {el1:13.4f}")
    print(f"error, N = {N2:3.0f}:\t {eu2:6.4f} | {el2:13.4f}")
    print(f"order of conv.:\t {ou:6.0f} | {ol:13.0f}")
    
    plt.plot(x1, U1[:, i1], "C0", label=f"Upwind, N={N1}")
    plt.plot(x2, U2[:, i2], "C4", label=f"Upwind, N={N2}")
    plt.plot(x1, L1[:, i1], "C1", label=f"Lax-Friedrich, N={N1}")
    plt.plot(x2, L2[:, i2], "C3", label=f"Lax-Friedrich, N={N2}")
    plt.plot(x2, exact_vals2, "dimgray", label="exact")
    plt.legend()
    plt.title(f"T = {T}")
    plt.show()
